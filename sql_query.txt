INSERT INTO Table4(InternalNumber, NameSurname, Position, SalaryMonth, Tax, Month) SELECT Positions.InternalNumber, CONCAT(Employees.Name, " ", Employees.Surname), Positions.Position, Employees.SalaryYear/12, SUM(Taxes.Taxes), Taxes.Month FROM Employees, Taxes, Positions where Taxes.EmployeeID = Employees.EmployeeID AND Positions.EmployeeID = Employees.EmployeeID group by Taxes.EmployeeID;
